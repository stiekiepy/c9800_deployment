! ### CREATE POLICY PROFILE AND WLAN PROFILE ###
!
! ### POLICY PROFILES
! 

wireless profile policy DEO-POLICY-PR
 description "DEO data WLAN Policy Profile"
 dhcp-tlv-caching
 http-tlv-caching
 ipv4 dhcp required
 vlan nem-wrl-usr
 no shutdown

wireless profile policy DEO_CWA_LOCAL_INET_PR
 aaa-override
 accounting-list radAccMethod
 description "Local Internet Connection"
 nac
 vlan nem-guestwifi
 no shutdown


! ### WLAN PROFILES ###
!
! # NEM-WRL-USR
!
aaa attribute list wlan_lobby_access

wlan NEM-WRL-USR 1 NEM-WRL-USR
 ccx aironet-iesupport
 radio dot11a
 security dot1x authentication-list radAutheMethod
 no shutdown

!
! # NEM-GuestWiFi
!
wlan NEM-GuestWiFi 2 NEM-GuestWiFi
 band-select
 mac-filtering radAutheZMethod
 no security ft adaptive
 no security wpa
 no security wpa wpa2
 no security wpa wpa2 ciphers aes
 no security wpa akm dot1x
 no shutdown

! # ADD WLAN- AND POLICY-PROFILES TO POLICY TAG

wireless tag policy DEO-POLICY-TAG
 description "Local policy tag for WLANs"
 wlan NEM-WRL-USR policy DEO-POLICY-PR
 wlan NEM-GuestWiFi policy DEO_CWA_LOCAL_INET_PR
